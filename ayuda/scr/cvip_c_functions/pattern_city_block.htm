NAME

        pattern_city_block – calculates the city block distance
 

SYNOPSIS

#include "stdio.h"
#include "math.h"
#include "float.h"
#include "stdlib.h"
#include "string.h"
      #define MAX_CASE 128
 
      int pattern_city_block(char *file_tt, char *file_tr, int *err_code)      
 
<file_tt> - pointer to the test text file
<file_tr> - pointer to the training text file
<err_code> - pointer to the error code; if no error use 0
 
 
 

PATH

        $CVIPtoolsHOME\CVIPC\FEATURE\pattern_city_block.c
 

DESCRIPTION

     This function calculates the city block distance measure from
     the elements of the two text files. The results are saved in a 
     temporary file named CVIPtemp. This temporary file can be found in the 
     system TEMP folder. The formula used for the calculation is 
     given in chapter 6 of the following book:
 
        Computer Imaging: Digital Image Analysis and Processing
 
           Scott E Umbaugh, A CRC Press Book, 2005
 
 

EXAMPLE

#include      "stdlib.h"

#include      "CVIPtoolkit.h"

#include      "CVIPpattern.h"

 

#define MAX_CASE 128

 

void main()

{

       char   *test, *training;

 

       /* pattern_city_block will return number of entries in training set */

       int           no_of_entries_tr;

 

       /*  Define a pointer pointing to the error code

              Initialize  it to 0.

       */

       int           *err_code;

       err_code = malloc(sizeof(int));

       *err_code = 0;

      

       /* Memory allocation */

       test = malloc(MAX_CASE*sizeof(char));

       training = malloc(MAX_CASE*sizeof(char));

 

       /* Get the file names, and parameters. */

       print_CVIP("\t\tPlease type in the file name for testing set: ");

       scanf("%s", test);

       print_CVIP("\t\tPlease type in the file name for training set: ");

       scanf("%s", training);

 

       /* Call c function, and return the err_code */

       no_of_entries_tr = pattern_city_block(test, training, err_code);

      

       /* If err_code is not 0, print out the err_code */

       if(*err_code != 0)

       {

              printf("Error code: %d\n", *err_code);

       }     

 

       /* Free pointers */

       free(test);

       free(training);

}

 

TYPES AND CONSTANTS

     None
 

RETURN VALUES

     Number of entries in the training set
