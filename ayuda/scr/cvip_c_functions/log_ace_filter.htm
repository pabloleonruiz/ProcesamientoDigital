NAME

     log_ace_filter - an Adaptive Contrast Enhancement filter
 

SYNOPSIS

      #include <CVIPtoolkit.h>
      #include <CVIPconvert.h>
      #include <CVIPdef.h>
      #include <CVIPimage.h>
      #include <CVIPspfltr.h>
 
     Image   *log_ace_filter(Image   *inputIMAGE,int   size,float
     alpha,float beta)
 
      <inputImage> - pointer to an Image
      <size> - mask size (3,5,7,9,...)
      <alpha> - local mean
      <beta> - local gain factor
 

PATH

     $CVIPtoolsHOME\CVIPC\SPATIALFILTER\lee_filter.c
 

DESCRIPTION

     log_ACE's algorithms are able to enhance  the  contrast  and
     dynamic  range of the image. log_ACE's algorithm uses linear
     brightness stretching to modify  the  dynamic  range  of  an
     image.  For  more  detail, see reference "The Study of Loga-
     rithmic Image Processing Model and Its Application to  Image
     Enhancement" by G.Deng, L.W. Cahill and G.R. Tobin.
 
      output = k1*[ln(Ibar(r,c)-ln(m_lbar(r,c))] + k2*m_lbar(r,c)
 
      where:
      m_lbar = 1-m_l(r,c)/M, normalized complement of local mean
      Ibar = 1 - I(r,c)/M, normalized complement of image
      M = number of gray levels (255)
      k1 = Local gain factor (multiplier), between 0 and 1
      k2 = Local mean factor (multiplier), between 0 and 1
      kernel size = size of local window
 

TYPES AND CONSTANTS

     None
 

RETURN VALUES

     Pointer to the processed image
 

HISTORY

     History information recorded: None
 

EXAMPLE

      #include <CVIPtoolkit.h>
      #include <CVIPconvert.h>
      #include <CVIPdef.h>
      #include <CVIPimage.h>
      #include <CVIPspfltr.h>
 
      void main() {
      Image *cvipImage;
      IMAGE_FORMAT format;
      char *inputfile,*outputfile;
 
      (void) setDisplay_Image(VIEWER, "Default");
      print_CVIP("\n\t\tEnter the  Input File Name:  ");
      inputfile =(char *) getString_CVIP();
      format = getFormat_CVIP(inputfile);
      cvipImage = read_Image(inputfile,1);
      view_Image(cvipImage, inputfile);
      free(inputfile);
      /* the following call performs filter operation in a
      kernel size of 5 with the local gain factor 0.5 multiplier
      and local mean multiplier equal to 1.0 */
      cvipImage = log_ace_filter(cvipImage, 5, 1.0, 0.5);
      print_CVIP("\n\t\tEnter the Output File Name:  ");
      outputfile = getString_CVIP();
      view_Image(cvipImage,outputfile);
      write_Image(cvipImage,outputfile,CVIP_NO,CVIP_NO,format,1);
      free(outputfile);
      }
