NAME

     kirsch_filter - perform kirsch edge detection
 

SYNOPSIS

      #include "CVIPimage.h"
      #include "CVIPdef.h"
      #include "CVIPfs.h"
      #include <limits.h>
 
     Image *kirsch_filter(Image *inputImage,  Image *dirImage, 
        int mask_choice,  int mask_size, int keep_dc, int threshold)
 
      <inputImage> - pointer to an Image
      <dirImage> - pointer to direction Image
      <mask_choice> - type of smoothing filter
      <mask_size> - kernel size
      <keep_dc> - 0 (no) or 1 (yes)
      <threshold> - value for binary threshold
 

PATH

     $CVIPtoolsHOME\CVIPC\SPATIALFILTER\kirsch_filter.c
 

DESCRIPTION

     This function applies kirsch edge detection algorithm to
     <inputImage>, and returns magnitude image, but for direction image, 
     a blank image is created using new_Image and is passed to 
     the function as direction image. After the function is
     called direction image can be then viewed using view_Image
     function. Data range of output direction image is -3pi/4 to +pi.
     If a smoothing filter is desired as a preprocessing step,
     set <mask_choice> to:
                       1 = Gaussian blur
                       2 = generic lowpass 1
                       3 = generic lowpass 2
                       4 = neighborhood average
 
     For <mask_size>: Set mask_size to desired kernel size
     For <dirImage> : direction image.
 
     NOTE: for the parameters, mask_choice, mask_size, keepdc, 
     threshold, use the value -1 for any parameter not needed. 
 

TYPES AND CONSTANTS

     None
 

RETURN VALUES

     A kirsch edge-detected magnitude image and direction image.     
 

HISTORY

     History information recorded: None
 

EXAMPLE

      #include <CVIPtoolkit.h>
      #include <CVIPimage.h>
      #include <CVIPdef.h>
      #include <CVIPfs.h>
      #include <CVIPconvert.h>
      #include <CVIPview.h>
      #define VIEWER "picture"
      #define VIDEO_APP "SunVideo &"
 
      void main()
      {
      Image *inputImage, *dirImage;
      Image *outImage;
      IMAGE_FORMAT format;
      char *inputfile,*outputfile,*outputfile1;
        
      print_CVIP("\n\t\tEnter the  Input File Name:  ");
      inputfile =(char *) getString_CVIP();
      format = getFormat_CVIP(inputfile);
      inputImage = read_Image(inputfile, 1);
      view_Image(inputImage, inputfile);      
      free(inputfile);
 

      dirImage = new_Image(inputImage->image_format,

      inputImage->color_space, inputImage->bands, getNoOfRows_Image(inputImage),

      getNoOfCols_Image(inputImage), CVIP_FLOAT, REAL);

     

      outImage=kirsch_filter(inputImage, dirImage, -1, -1, -1, -1);

      
      print_CVIP("\n\t\tEnter the Output File Name of magnitude Image:  ");
      outputfile = getString_CVIP();
      print_CVIP(“\n\t\tEnter the Output File Name of direction Image: “);
      outputfile1 = getString_CVIP();
      view_Image(outImage,outputfile);
      view_Image(dirImage, outputfile1);
      delete_Image(outImage);
      delete_Image(dirImage);
      delete_Image(inputImage);
      free(outputfile);
      free(outputfile1);
      }
