NAME

     harris_filter - a spatial- a corner detection filter.
 

SYNOPSIS

      #include <CVIPtoolkit.h>
      #include <CVIPconvert.h>
      #include <CVIPdef.h>
      #include <CVIPimage.h>
      #include <CVIPspfltr.h>
 
     Image    *harris_filter(Image    *inputIMAGE,float    alpha,int
     threshold, int border,float dmin)
 
      <inputImage> - pointer to an Image
      <alpha> - sensitivity of the detector
      <threshold> - threshold value
      <border> -  the number of rows and colums along the image border
      <dmin> - radius of the circle to check for the corner
 
 

PATH

     $CVIPtoolsHOME\CVIPC\SPATIALFILTER\harris_filter.c
 

DESCRIPTION

     The Harris filter is a spatial-domain method for for corner detector developed by
     Harris and Stephens in 1988.In this filter corner response function to find the corners.
 

       

 
     Where CRF(r,c) represents the corners  of  the original  image  
     And G(P1*P1) is the Gaussian of the vertical gradient. G(P2*P2)
     Is the Gaussian of horizontal gradient.    

TYPES AND CONSTANTS

     None
 

RETURN VALUES

     Pointer to the processed image
 

HISTORY

     History information recorded: None
 

 

EXAMPLE

      #include <CVIPtoolkit.h>
      #include <CVIPconvert.h>
      #include <CVIPdef.h>
      #include <CVIPimage.h>
      #include <CVIPspfltr.h>
 
      void main() {
      Image *cvipImage;
      IMAGE_FORMAT format;
      char *inputfile,*outputfile;
 
      (void) setDisplay_Image(VIEWER, "Default");
      print_CVIP("\n\t\tEnter the  Input File Name:  ");
      inputfile =(char *) getString_CVIP();
      format = getFormat_CVIP(inputfile);
      cvipImage = read_Image(inputfile,1);
      view_Image(cvipImage, inputfile);
      free(inputfile);
      /* the following call performs filter operation in a
      With sensitivity alpha of 0.05, threshold 20000 , border value of 20 and
      the dmin value of 10*/
      cvipImage = harris_filter(cvipImage, 0.05f, 20000, 20, 10.0f);
      print_CVIP("\n\t\tEnter the Output File Name:  ");
      outputfile = getString_CVIP();
      view_Image(cvipImage,outputfile);
      write_Image(cvipImage,outputfile,CVIP_NO,CVIP_NO,format,1);
      free(outputfile);
      }
