NAME

     exp_ace_filter -  a  method  to  enhance  the  contrast  and
     dynamic range of the image
 

SYNOPSIS

      #include <CVIPtoolkit.h>
      #include <CVIPconvert.h>
      #include <CVIPdef.h>
      #include <CVIPimage.h>
      #include <CVIPspfltr.h>
 
 
      Image *exp_ace_filter(Image *inputIMAGE,int size,float beta,float alpha)
 
      <inputImage> - pointer to an Image
      <size> - mask size (3,5,7,9,...)
      <alpha> - local gain factor
      <beta> - local mean factor
 

PATH

     $CVIPtoolsHOME\CVIPC\SPATIALFILTER\lee_filter.c
 

DESCRIPTION

     exp_ACE's algorithms are able to enhance  the  contrast  and
     dynamic  range of the image. exp_ACE's algorithm uses linear
     brightness stretching to modify  the  dynamic  range  of  an
     image.
 
      output =  M*[I(r,c)/M]**k1 + [m_l(r,c)/I(r,c)]**k2
 
      where M =  the number of gray levels (255)
            m_l = the local mean
            k1 = Local gain factor (exponent)
            k2 = Local mean factor (exponent)
            kernel size = size of local window
 

TYPES AND CONSTANTS

     None
 

RETURN VALUES

     Pointer to the processed image
 

HISTORY

     History information recorded: None
 

EXAMPLE

      #include <CVIPtoolkit.h>
      #include <CVIPconvert.h>
      #include <CVIPdef.h>
      #include <CVIPimage.h>
      #include <CVIPspfltr.h>
 
      void main() {
      Image *cvipImage;
      IMAGE_FORMAT format;
      char *inputfile,*outputfile;
 
      (void) setDisplay_Image(VIEWER, "Default");
      print_CVIP("\n\t\tEnter the  Input File Name:  ");
      inputfile =(char *) getString_CVIP();
      format = getFormat_CVIP(inputfile);
      cvipImage = read_Image(inputfile,1);
      view_Image(cvipImage, inputfile);
      free(inputfile);
      /* the following call performs filter operation in a
      kernel size of 5 with the local gain factor 0.5 multiplier
      and local mean multiplier equal to 1.0 */
      cvipImage = exp_ace_filter(cvipImage, 5, 1.0, 0.5);
      print_CVIP("\n\t\tEnter the Output File Name:  ");
      outputfile = getString_CVIP();
      view_Image(cvipImage,outputfile);
      write_Image(cvipImage,outputfile,CVIP_NO,CVIP_NO,format,1);
      free(outputfile);
      }
