NAME

     cerchar_filter - a spatial- multi spectral image edge detection filter.
 

SYNOPSIS

      #include <CVIPtoolkit.h>
      #include <CVIPconvert.h>
      #include <CVIPdef.h>
      #include <CVIPimage.h>
      #include <CVIPspfltr.h>
 
     Image    Cerchar_Filter(Image* inputImage);     
      <inputImage> - pointer to an Image
      
 

PATH

     $CVIPtoolsHOME\CVIPC\SPATIALFILTER\cerchar_filter.c
 

DESCRIPTION

     Cervenka and Charvat Filter is used to find the edges in multispectral images. This can be applied to three band color images, as well as multispectral satellite images.

The result of this edge detector at pixel (r,c) is

  

 

 

Ib(r,c) is the arithmetic average of all the pixels in all bands at pixel location (r,c)  

 is the value at location (r,c) in the bth band, with a total of n bands

 

TYPES AND CONSTANTS

     None
 

RETURN VALUES

     Pointer to the processed image
 

HISTORY

     History information recorded: None
 

EXAMPLE

      #include <CVIPtoolkit.h>
      #include <CVIPconvert.h>
      #include <CVIPdef.h>
      #include <CVIPimage.h>
      #include <CVIPspfltr.h>
 
      void main() {
      Image *cvipImage;
      IMAGE_FORMAT format;
      char *inputfile,*outputfile;
 
      (void) setDisplay_Image(VIEWER, "Default");
      print_CVIP("\n\t\tEnter the  Input File Name:  ");
      inputfile =(char *) getString_CVIP();
      format = getFormat_CVIP(inputfile);
      cvipImage = read_Image(inputfile,1);
      view_Image(cvipImage, inputfile);
      free(inputfile);
      cvipImage = cerchar_filter(cvipImage);
      print_CVIP("\n\t\tEnter the Output File Name:  ");
      outputfile = getString_CVIP();
      view_Image(cvipImage,outputfile);
      write_Image(cvipImage,outputfile,CVIP_NO,CVIP_NO,format,1);
      free(outputfile);
      }
