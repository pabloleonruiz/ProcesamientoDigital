NAME

        nearest_centroid – finds the nearest centroid using a distance 
                           or similarity measure
 

SYNOPSIS

#include "math.h"
#include "stdio.h"
#include "stdlib.h"
#include "string.h"
#include "CVIPpattern.h"
     
int nearest_centroid(char *file_tt, char *file_tr, char *file_out, 
int option_normalize, int option_distance, float s_min, float s_max, 
float r_softmax, int r_minkowski)
      
<file_tt> - pointer to a text file (test set)
<file_tr> - pointer to a text file (training set)
<file_out> - pointer to a text file (output)
<option_normalize> - normalization method to be used
<option_distance> - distance or similarity method to be used
<s_min> - minimum value for the specified range (used with Min-max 
          normalization only; else use -1)
<s_max> - maximum value for the specified range (used with Min-max 
          normalization only; else use -1)
<r_softmax> - used with Softmax scaling only; else use -1
<r_minkowski> - used with Minkowski generalized distance measure
                only; else use -1
 
 
 
 

PATH

        $CVIPtoolsHOME\CVIPC\FEATURE\nearest_centroid.c
 

DESCRIPTION

     This function uses the selected distance or similarity measure
     to find the closest match in the training set to every entry in 
     the test set. It uses the training set to generate the centroid 
     for all the classes in the training file and then uses this 
     new training file as the main training file.
 
     Values to be used for:    
     
     option_normalize: 
        
        0 – No Normalization
        1 – Range-normalize
        2 – Unit Vector normalization
        3 – Standard Normal Density normalization
        4 – Min-max normalization
        5 – Softmax scaling
         
     option_distance: 
        
        1 – Euclidean Distance
        2 – City block or Absolute value metric
        3 – Maximum value metric
        4 – Minkowski distance
        5 – Vector inner product
        6 – Tanimoto metric
 

EXAMPLE

#include      "stdlib.h"

#include      "CVIPtoolkit.h"

#include      "CVIPpattern.h"

 

#define MAX_CASE 128

 

void main()

{

       char   *test, *training, *output;

       /*     Define all parameters, and initialize them to -1.

              Change them afterwards, if they are required in the

              application.

       */

       int           r_minkowski=-1, option_normalize=-1, option_distance=0;

       float  s_min=-1, s_max=-1, r_softmax=-1;

 

       int           err_code;

      

       /* Memory allocation */

       test = malloc(MAX_CASE*sizeof(char));

       training = malloc(MAX_CASE*sizeof(char));

       output = malloc(MAX_CASE*sizeof(char));

 

       /* Get the file names, and parameters. */

       print_CVIP("\t\tPlease type in the file name for testing set: ");

       scanf("%s", test);

       print_CVIP("\t\tPlease type in the file name for trainning set: ");

       scanf("%s", training);

       print_CVIP("\t\tPlease type in the file name for output: ");

       scanf("%s", output);

 

       /* Set required parameters */

       option_normalize = 1;

       option_distance = 0;

 

       /* Call c function, and return the err_code */

err_code = nearest_centroid(test, training, output, option_normalize, option_distance, s_min, s_max, r_softmax, r_minkowski);

      

       /* If err_code is not 0, print out the err_code */

       if(err_code != 0)

       {

              printf("Error code: %d\n", err_code);

       }     

 

       /* Free pointers */

       free(test);

       free(training);

       free(output);

}

 

TYPES AND CONSTANTS

     None
 

RETURN VALUES

     An error code
 
 Error Codes:
 
1: The headers in the training and test files don't match 
2: K is greater than the number of entries in the training setor K<=0 
3: s_min is greater than or equal to s_max 
4: Data range is 0 and so the data can't be normalized 
5: Divisor is 0 in Tanimoto similarity measure 
7: Can't open, read or create file 
9: Memory allocation error 
10: Every entry in training set should have a class name 
12: r in minkowski should be a positive integer 
13: There is more than one header in the training or test feature file
