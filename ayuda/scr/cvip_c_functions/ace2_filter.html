NAME

     ace2_filter - a spatial- an Adaptive Contrast  and  Enhance-
     ment filter
 

SYNOPSIS

      #include <CVIPtoolkit.h>
      #include <CVIPconvert.h>
      #include <CVIPdef.h>
      #include <CVIPimage.h>
      #include <CVIPspfltr.h>
 
     Image    *ace2_filter(Image    *inputIMAGE,int    size,float
     alpha,float beta)
 
      <inputImage> - pointer to an Image
      <size> - mask size (3,5,7,9,...)
      <alpha> - local mean
      <beta> - local gain factor
 

PATH

     $CVIPtoolsHOME\CVIPC\SPATIALFILTER\lee_filter.c
 

DESCRIPTION

     The ACE2 filter is a spatial-domain method for contrast  and
     dynamic  range  modifications with less limitation on linear
     contrast stretching. For more  detail,  see  "Digital  image
     enhancement  and noise filtering by use of local statistics"
     by J.S.Lee
 
        ACE2_filter = K2 * m_l(r,c) + K1 * [I(r,c) - m_l(r,c)]
 
     Where I(i,j) represents the pixel brightness  value  of  the
     original  image  and m_l(r,c) is the arithmetic mean bright-
     ness value of an (n x n) window  that  is  centered  on  the
     pixel position (r,c).  The parameters "K1" = local gain fac-
     tor and "K2" = local mean factor.
 

TYPES AND CONSTANTS

     None
 

RETURN VALUES

     Pointer to the processed image
 

HISTORY

     History information recorded: None
 

EXAMPLE

      #include <CVIPtoolkit.h>
      #include <CVIPconvert.h>
      #include <CVIPdef.h>
      #include <CVIPimage.h>
      #include <CVIPspfltr.h>
 
      void main() {
      Image *cvipImage;
      IMAGE_FORMAT format;
      char *inputfile,*outputfile;
 
      (void) setDisplay_Image(VIEWER, "Default");
      print_CVIP("\n\t\tEnter the  Input File Name:  ");
      inputfile =(char *) getString_CVIP();
      format = getFormat_CVIP(inputfile);
      cvipImage = read_Image(inputfile,1);
      view_Image(cvipImage, inputfile);
      free(inputfile);
      /* the following call performs filter operation in a
      kernel size of 5 with the local gain factor 0.5 multiplier
      and local mean multiplier equal to 1.0 */
      cvipImage = ace2_filter(cvipImage, 5, 1.0, 0.5);
      print_CVIP("\n\t\tEnter the Output File Name:  ");
      outputfile = getString_CVIP();
      view_Image(cvipImage,outputfile);
      write_Image(cvipImage,outputfile,CVIP_NO,CVIP_NO,format,1);
      free(outputfile);
      }
