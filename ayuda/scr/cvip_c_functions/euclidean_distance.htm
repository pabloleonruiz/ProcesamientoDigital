NAME

        euclidean_distance – calculates the euclidean distance
 

SYNOPSIS

#include "math.h"
     
float euclidean_distance(float *vector1, float *vector2, int n)
      
<vector1> - pointer to an input array
<vector2> - pointer to an input array
<n> - number of elements (features) in the array
 
 
 

PATH

        $CVIPtoolsHOME\CVIPC\FEATURE\euclidean_distance.c
 

DESCRIPTION

     This function calculates the Euclidean distance measure from
     the elements of the two arrays. The formula used for the 
     calculation is given in chapter 6 of the following book:
 
        Computer Imaging: Digital Image Analysis and Processing
 
           Scott E Umbaugh, A CRC Press Book, 2005
 
 

EXAMPLE

#include      "stdlib.h"

#include      "CVIPtoolkit.h"

#include      "CVIPpattern.h"

 

#define MAX_CASE 3

 

void main()

{

       float  *vector1;

       float  *vector2;

       float  result;

       int           i;

 

       vector1 = malloc(MAX_CASE*sizeof(float));

       vector2 = malloc(MAX_CASE*sizeof(float));

 

       /* Get the vectors. The items should be separated by "," */

       print_CVIP("\t\tPlease type in the items in vector1: ");

       for(i=0;i<MAX_CASE;i++)

       {

              scanf("%f,", &vector1[i]);

       }

      

       print_CVIP("\t\tPlease type in the items in vector1: ");

       for(i=0;i<MAX_CASE;i++)

       {

              scanf("%f,", &vector2[i]);

       }

 

       /* Call c function, and return the err_code */

       result = euclidean_distance(vector1, vector2, MAX_CASE);

      

       /* Print out the result */

       printf("result=%f\n",result);

 

       /* Free the pointers */

       free(vector1);

       free(vector2);

}

 

TYPES AND CONSTANTS

     None
 

RETURN VALUES

     A float number representing the Euclidean distance
