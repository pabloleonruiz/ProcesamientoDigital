NAME

     Conversion- CVIPtools  core  toolbox  library  containing
     routines for image file I/O and format conversion
 

SYNOPSIS

      #include <CVIPtoolkit.h>
      #include <CVIPdef.h>
      #include <CVIPimage.h>
      #include <CVIPconvert.h>
 
     See the individual function man page for details  and  exam-
     ples.
 

PATH

  $CVIPtoolsHOME\lib
   
 

TYPES AND CONSTANTS

     typedef enum {PBM, PGM, PPM, EPS, TIF, GIF, RAS, ITX,IRIS, CCC, BIN, BMP, VIP, GLR, BTC, BRC, HUF, LZW, ARITH, BTC2, BTC3, DPC, SAFVR} IMAGE_FORMAT
